436  public static ByteBuffer packIt(byte[] bytes, boolean compress)
437  {
438    /*
439      Setting up the protocol header. This is 4 bytes long
440      represented as an integer. The first 2 bits indicate
441      the serializer type. The 3rd bit indicates if compression
442      is turned on or off. It is turned off by default. The 4th
443      bit indicates if we are in streaming mode. It is turned off
444      by default. The 5th-8th bits are reserved for future use.
445      The next 8 bits indicate a version number. Remaining 15 bits
446      are not used currently.
447    */
448    int header = 0;
449    // Setting up the serializer bit
450    header |= serializerType_.ordinal();
451    // set compression bit.
452    if (compress)
453      header |= 4;
454    // Setting up the version bit
455    header |= (version_ << 8);
456 
457    ByteBuffer buffer = ByteBuffer.allocate(4 + 4 + 4 + bytes.length);
458    buffer.putInt(PROTOCOL_MAGIC);
459    buffer.putInt(header);
460    buffer.putInt(bytes.length);
461    buffer.put(bytes);
462    buffer.flip();
463    return buffer;
464  }
