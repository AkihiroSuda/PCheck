665    {

667        {

669            if ( ep.equals( localEndPoint_ ) )

682

698                        }

811    /* Send all the data with version greater than maxRemoteVersion */
812    void sendAll(GossipDigest gDigest, Map<InetAddress, EndPointState> deltaEpStateMap, int maxRemoteVersion)
813    {
814        EndPointState localEpStatePtr = getStateForVersionBiggerThan(gDigest.getEndPoint(), maxRemoteVersion) ;
815        if ( localEpStatePtr != null )
816            deltaEpStateMap.put(gDigest.getEndPoint(), localEpStatePtr);
817    }
818 
819    /*
820        This method is used to figure the state that the Gossiper has but Gossipee doesn't. The delta digests
821        and the delta state are built up.
822    */
823    synchronized void examineGossiper(List<GossipDigest> gDigestList, List<GossipDigest> deltaGossipDigestList, Map<InetAddress, EndPointState> deltaEpStateMap)
824    {
825        for ( GossipDigest gDigest : gDigestList )
826        {
827            int remoteGeneration = gDigest.getGeneration();
828            int maxRemoteVersion = gDigest.getMaxVersion();
829            /* Get state associated with the end point in digest */
830            EndPointState epStatePtr = endPointStateMap_.get(gDigest.getEndPoint());
831            /*
832                Here we need to fire a GossipDigestAckMessage. If we have some data associated with this endpoint locally
833                then we follow the "if" path of the logic. If we have absolutely nothing for this endpoint we need to
834                request all the data for this endpoint.
835            */
836            if ( epStatePtr != null )
837            {
838                int localGeneration = epStatePtr.getHeartBeatState().getGeneration();
839                /* get the max version of all keys in the state associated with this endpoint */
840                int maxLocalVersion = getMaxEndPointStateVersion(epStatePtr);
841                if ( remoteGeneration == localGeneration && maxRemoteVersion == maxLocalVersion )
842                    continue;
843 
844                if ( remoteGeneration > localGeneration )
845                {
846                    /* we request everything from the gossiper */
847                    requestAll(gDigest, deltaGossipDigestList, remoteGeneration);
848                }
849                if ( remoteGeneration < localGeneration )
850                {
851                    /* send all data with generation = localgeneration and version > 0 */
852                    sendAll(gDigest, deltaEpStateMap, 0);
853                }
854                if ( remoteGeneration == localGeneration )
855                {
856                    /*
857                        If the max remote version is greater then we request the remote endpoint send us all the data
858                        for this endpoint with version greater than the max version number we have locally for this
859                        endpoint.
860                        If the max remote version is lesser, then we send all the data we have locally for this endpoint
861                        with version greater than the max remote version.
862                    */
863                    if ( maxRemoteVersion > maxLocalVersion )
864                    {
865                        deltaGossipDigestList.add( new GossipDigest(gDigest.getEndPoint(), remoteGeneration, maxLocalVersion) );
866                    }
867                    if ( maxRemoteVersion < maxLocalVersion )
868                    {
869                        /* send all data with generation = localgeneration and version > maxRemoteVersion */
870                        sendAll(gDigest, deltaEpStateMap, maxRemoteVersion);
871                    }
872                }
873            }
874            else
875            {
876                /* We are here since we have no data for this endpoint locally so request everything. */
877                requestAll(gDigest, deltaGossipDigestList, remoteGeneration);
878            }
879        }
880    }

1008 
1009        /*
1010         * First construct a map whose key is the endpoint in the GossipDigest and the value is the
1011         * GossipDigest itself. Then build a list of version differences i.e difference between the
1012         * version in the GossipDigest and the version in the local state for a given InetAddress.
1013         * Sort this list. Now loop through the sorted list and retrieve the GossipDigest corresponding
1014         * to the endpoint from the map that was initially constructed.
1015        */
1016        private void doSort(List<GossipDigest> gDigestList)
1017        {
1018            /* Construct a map of endpoint to GossipDigest. */
1019            Map<InetAddress, GossipDigest> epToDigestMap = new HashMap<InetAddress, GossipDigest>();
1020            for ( GossipDigest gDigest : gDigestList )
1021            {
1022                epToDigestMap.put(gDigest.getEndPoint(), gDigest);
1023            }
1024 
1025            /*
1026             * These digests have their maxVersion set to the difference of the version
1027             * of the local EndPointState and the version found in the GossipDigest.
1028            */
1029            List<GossipDigest> diffDigests = new ArrayList<GossipDigest>();
1030            for ( GossipDigest gDigest : gDigestList )
1031            {
1032                InetAddress ep = gDigest.getEndPoint();
1033                EndPointState epState = Gossiper.instance.getEndPointStateForEndPoint(ep);
1034                int version = (epState != null) ? Gossiper.instance.getMaxEndPointStateVersion( epState ) : 0;
1035                int diffVersion = Math.abs(version - gDigest.getMaxVersion() );
1036                diffDigests.add( new GossipDigest(ep, gDigest.getGeneration(), diffVersion) );
1037            }
1038 
1039            gDigestList.clear();
1040            Collections.sort(diffDigests);
1041            int size = diffDigests.size();
1042            /*
1043             * Report the digests in descending order. This takes care of the endpoints
1044             * that are far behind w.r.t this local endpoint
1045            */
1046            for ( int i = size - 1; i >= 0; --i )
1047            {
1048                gDigestList.add( epToDigestMap.get(diffDigests.get(i).getEndPoint()) );
1049            }
1050        }
1051    }
